select * 
from AddPlayer t1
left outer join AddTeam t2 on t2.team_name = t1.team_name
left outer join user t3 on t3.username = t2.username
where t3.username = User.username




select map.first_name, map.last_name, map.contact, mat.team_name
from auth_user au
left outer join manager_addteam mat on mat.username_id = au.id
left outer join manager_addplayer map on map.team_name_id = mat.id
where au.id = 1
